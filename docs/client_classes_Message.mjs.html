/**
 * @class
 * @category Client
 */
class Message {
	constructor(oAttr={}) 
	{
		oAttr = {...{
			owner: "customer",
			type: "text",
			date: (+ new Date()) / 1000,
			id: -1 
		}, ...oAttr}

		this.owner = oAttr.owner 
		this.type = oAttr.type 
		this.text = oAttr.text 
		this.date = oAttr.date 
		this.id = oAttr.id 
		this.url = oAttr.url 
		this.size = oAttr.size 
		this.duration = oAttr.duration 
		this.name = oAttr.name 
	}
}

export default Message
